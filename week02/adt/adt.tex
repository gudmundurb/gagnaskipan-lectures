\begin{frame}{Hugræn gagnatög}
    \begin{itemize}
        \item \emph{Tag} er safn (mengi) gilda
            \begin{itemize}
                \item Boolean-tagið (\cpp|bool|) samanstendur af gildunum \cpp|true| og \cpp|false|
            \end{itemize}
        \item \emph{Gagnatag} er tag með aðgerðum sem hægt er að beita á meðlimi tagsins
            \begin{itemize}
                \item Heiltölur (\cpp|int|) ásamt reikniaðgerðum eru gagnatag
            \end{itemize}
        \item \emph{Hugrænt gagnatag} (e. Abstract data type) er skilgreining á
            gagnatagi sem er eingöngu lýst sem gagnaliðum og aðgerðum sem hægt
            er að beita á þá
    \end{itemize}
\end{frame}

\begin{frame}{Gagnaskipan}
    \begin{itemize}
        \item Gagnaskipan (e. data structure) er útfærsla á hugrænu gagnatagi
        \item Í hlutbundnum (e. object oriented) forritunarmálum eins og C++ eru hugræn gagnatög útfærð sem klasar
            \begin{itemize}
                \item Gagnaliðirnir eru meðlimabreytur
                \item Aðgerðirnar eru klasaföll
            \end{itemize}
        \item Klasar hjálpa okkur með \emph{gagnahuld} (e. information hiding)
            \begin{itemize}
                \item Sá sem notar gagnaskipan þarf ekki að þekkja útfærsluna
                \item \dots eða hvað?
            \end{itemize}
    \end{itemize}
\end{frame}
