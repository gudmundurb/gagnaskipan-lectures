\begin{frame}{Frávik}
    \begin{itemize}
        \item Frávik (e. exception) gefa okkur leið til að taka á fráviki frá eðlilegri hegðun forrits
        \item Frávikum er \emph{kastað} (e. thrown)
        \item Notum lykilorðið \cpp|throw| til að kasta fráviki (villu)
        \item Hægt er að \emph{grípa} (e. catch) villur með \cpp|try| og \cpp|catch|
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Dæmi}
    \begin{lstlisting}
#include <iostream>
using namespace std;

int main () {
    try {
        throw 20;
    }
    catch (int e) {
        cout << "Exception Nr. " << e << "occurred\n";
    }
    return 0;
}
    \end{lstlisting}
\end{frame}

\begin{frame}{Fráviksklasar}

        \item Klasarnir segja til um \emph{hvað} gerðist
    \end{itemize}
