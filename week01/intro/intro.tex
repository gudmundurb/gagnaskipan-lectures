\begin{frame}{Mooshak}
    \begin{itemize}
        \item \textbf{Öllum verkefnum námskeiðsins þarf að skila inn á Mooshak}
        \item Farið verður yfir flest verkefni í höndunum
        \item Vægi Mooshak og yfirferðar dæmatímakennara (ef við á) kemur fram á verkefnum
        \item Aðeins verður farið yfir þau verkefni sem skilað er inn á Mooshak og þýðast
        \item \textbf{Síðustu} samþykktu skil sem send eru inn á Mooshak teljast vera skil á verkefni
            \begin{itemize}
                \item Hægt að laga kóðann sinn eftir að maður fær samþykkt
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Skil á Mooshak}
    Fjöldi skila inn á Mooshak getur haf áhrif á einkunn
    \begin{itemize}
        \item Sérhvert verkefni hefur ákveðinn fjölda \emph{frírra skila}
        \item Eftir að öll frí skil hafa verið notuð dregst 0,05 frá einkunn við hver skil
        \item "`Compile Time Error"' telst ekki sem skil
        \item Frádrátturinn er reiknaður fram að \textbf{fyrstu} samþykktu skilum, eftir það eru öll skil frí
        \item Frádrátturinn verður aldrei meiri en 3,0
    \end{itemize}
\end{frame}

\begin{frame}{Reglur um notkun Mooshak}
    \begin{itemize}
        \item Lausn sem send er inn á Mooshak telst vera hugverk þess sem sendir hana inn
        \item Þið megið aðeins senda inn kóða sem þið skrifið sjálf (fyrir utan grunnskjöl sem fylgja verkefnum)
        \item Ekki má nota Mooshak til að "`prófa"' kóða af netinu eða kóða frá samnemendum
    \end{itemize}
\end{frame}

\begin{frame}{Verkefnisfundur}
    Til að standa vörð um heiðarleika verkefnavinnu í námskeiðinu, þá áskilur kennari sér rétt til að boða alla þá nemendur, sem skila verkefni til yfirferðar, á verkefnisfund.
    \begin{itemize}
        \item Fundarboðið mun ekki berast seinna en þremur vikum eftir að verkefni hefur verið skilað
        \item Svari nemandi ekki fundarboði, eða mæti ekki á fund án skýringar, fær nemandi 0 fyrir verkefnið þangað til að verkefnisfundur hefur verið haldinn
        \item Á fundinum verður nemandi beðinn um að lýsa lausn sinni og hvernig hún virkar
        \item Sýni nemandi ekki nægan skilning á lausn sinni, áskilur kennari sér rétt til að lækka verkefniseinkunn
    \end{itemize}
\end{frame}
