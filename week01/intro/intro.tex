\section{Námsmat}

\begin{frame}{Verkefni}
  \begin{itemize}
    \item Skilaverkefni (25\%)
      \begin{itemize}
        \item Fimm stærri skilaverkefni yfir önnina
        \item Hvert gildir 5\%
      \end{itemize}
    \item Dæmatímaverkefni (10\%)
      \begin{itemize}
        \item 11 minni verkefni yfir önnina
        \item 8 bestu af 11 gilda til einkunnar
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Lokapróf}
  \begin{itemize}
    \item Gildir 65\% af lokaeinkunn
    \item Mun að öllum líkindum vera bæði rafrænt og skriflegt
    \item Skriflega prófið verður líklegast seint á önninni
    \item Ná þarf að lágmarki 5,0 í báðum hlutum prófsins til að standast prófið
  \end{itemize}
\end{frame}

\begin{frame}{Aukaverkefni}
  \begin{itemize}
    \item Aukaverkefni sem má vinna hvenær sem er yfir önnina
    \item Lítil aukaverkefni
      \begin{itemize}
        \item Svipuð dæmatímaverkefnum
        \item Gilda til hækkunar á verkefnaeinkunn
        \item Geta hækkað verkefnaeinkunn um allt að 5\%
      \end{itemize}
    \item Stórt aukaverkefni
      \begin{itemize}
        \item Getur komið í stað eins skilaverkefnis
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Verkefnaskil}
  \begin{itemize}
    \item Skilaverkefni
      \begin{itemize}
        \item Skilafrestur er kl. 23:59 á mánudögum í viku 4, 6, 8, 10 og 12
      \end{itemize}
    \item Dæmatímaverkefni
      \begin{itemize}
        \item Lögð fyrir á mánudögum
        \item Skilafrestur er kl. 23:59 á fimmtudögum
          \begin{itemize}
            \item Undandtekning: Í þessari viku er skilafresturinn á sunnudag
          \end{itemize}
        \item Ekki dæmatímaverkefni í síðustu viku
      \end{itemize}
    \item Aukaverkefni
      \begin{itemize}
        \item Lögð fyrir reglulega yfir önnina
        \item Skilafrestur er fram til lokaprófs
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Verkefnaskil}
  \begin{itemize}
    \item Öllum verkefnum á að skila fyrir 23:59 á skiladegi
    \item Einkunn lækkar um einn heilan fyrir hvern dag sem skilað er of seint
    \item Helgardagar og frídagar eru meðtaldir
    \item Ekki er tekið við verkefnum sem skilað er meira en tveimur dögum of seint
  \end{itemize}
\end{frame}

\begin{frame}{Mooshak}
  \begin{itemize}
    \item \textbf{Öllum verkefnum námskeiðsins þarf að skila inn á Mooshak}
    \item Að auki verður farið yfir sum verkefni í höndunum
    \item Vægi Mooshak og yfirferðar dæmatímakennara (ef við á) kemur fram á verkefnum
    \item Aðeins verður farið yfir þau verkefni sem skilað er inn á Mooshak og þýðast
  \end{itemize}
\end{frame}

\begin{frame}{Reglur um notkun Mooshak}
  \begin{itemize}
    \item Lausn sem send er inn á Mooshak telst vera hugverk þess sem sendir hana inn
    \item Þið megið aðeins senda inn kóða sem þið skrifið sjálf (fyrir utan grunnskjöl sem fylgja verkefnum)
    \item Ekki má nota Mooshak til að "`prófa"' kóða af netinu eða kóða frá samnemendum
  \end{itemize}
\end{frame}

\begin{frame}{Utanaðkomandi aðstoð}
  \begin{itemize}
    \item Að afrita lausn sem þið finnið á netinu, eða fáið frá samnemanda, og skila inn sem eigin lausn er \textbf{ritstuldur}
    \item Það er brot á Siðareglum HR (\url{http://www.ru.is/haskolinn/sidareglur-hr/})
      \begin{itemize}
        \item "`Við virðum hugverkaréttindi, eignum okkur ekki heiðurinn af hugmyndum annarra, og getum ávallt þeirra heimilda sem við notum, í samræmi við venjur vísindasamfélagsins."'
      \end{itemize}
    \item "`Google your work, not the problem"'
  \end{itemize}
\end{frame}

\begin{frame}{Viðurlög við brotum}
  \begin{itemize}
    \item Skólinn lítur brot á Náms- og prófareglum alvarlegum augum
      \begin{itemize}
        \item Fyrsta brot: 0 fyrir verkefni
        \item Annað brot: 0 fyrir námskeið
        \item Þriðja brot: Brottrekstur úr skóla
      \end{itemize}
    \item Fyrirgerið rétti til skiptináms
    \item Fáið ekki að koma fram fyrir hönd skólans
  \end{itemize}
\end{frame}

\section{Að lokum}

\begin{frame}{Hvað er gagnaskipan?}
  \begin{itemize}
    \item Gagnaskipan (data structure) er ákveðin leið til að geyma gögn og skipuleggja framsetningu þeirra
    \item Dæmi um einföld gagnaskipan sem þið hafið séð
      \begin{itemize}
        \item Fylki
        \item Struct
        \item Klasar
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Af hverju gagnaskipan?}
  \begin{itemize}
    \item Sem tölvunarfræðingar erum við alltaf að vinna með gögn
      \begin{itemize}
        \item Magn gagna í heiminum eykst stöðugt
        \item Google veit af 30 billjón vefsíðum (30.000.000.000.000)
      \end{itemize}
    \item Þurfum að þekkja kosti, galla og sklivirkni ýmissa framsetninga gagna
    \item Í leiðinni bætum við forritunarþekkingu
  \end{itemize}
\end{frame}
